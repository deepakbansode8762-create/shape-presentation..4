CIRCLE CLASS
class Circle extends Shape { private double radius; public Circle(double radius) { this.radius = radius; } @Override public double calculateArea() { return Math.PI * radius * radius; } @Override public double calculatePerimeter() { return 2 * Math.PI * radius; } }

TRIANGLE CLASS
class Triangle extends Shape { private double side1, side2, side3; public Triangle(double s1, double s2, double s3) { this.side1 = s1; this.side2 = s2; this.side3 = s3; } @Override public double calculateArea() { double s = (side1 + side2 + side3) / 2; return Math.sqrt(s * (s - side1) * (s - side2) * (s - side3)); } @Override public double calculatePerimeter() { return side1 + side2 + side3; } }
